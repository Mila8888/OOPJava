package task1;

public class RobotClass implements Participants {

    private String name;
    private int Limit;
    private int Power;

    public RobotClass(String name, int limit, int power) {
        this.name = name;
        Limit = limit;
        Power = power;
        boolean isOnDistance = false;
    }

    @Override
    public String name() {
        return null;
    }

    @Override
    public void run(int length) {
        if (length <= Limit) {
            System.out.println("Робот " + name  + ": "+ OnDistance );
        }else {
            System.out.println("Робот " + name + ": " + isOnDistance );
        }
    }

    @Override
    public void jump(int height) {
        if (height <= Power) {
            System.out.println("Робот " + name + ": " + OnDistance );
        }else {
            System.out.println("Робот " + name + ": " + isOnDistance );
        }
    }


    @Override
    public void info() {
        System.out.println("Робот " + name + " MAX пробежал " + Limit + " км" + " и MAX прыгнул " + Power + " м");
    }


}
