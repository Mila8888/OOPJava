package task1;

public class HumanClass implements Participants {

        private String name;
        private int Limit;
        private int Power;

    public HumanClass(String name, int limit, int power) {
        this.name = name;
        Limit = limit;
        Power = power;
    }

    @Override
        public String name() {
            return null;
        }

        @Override
        public void run(int length) {
            if (length <= Limit) {
                System.out.println("Человек " + name + ": " + OnDistance );
            } else {
                System.out.println("Человек " + name + ": " + isOnDistance );
            }
        }

        @Override
        public void jump(int height) {
            if (height <= Power) {
                System.out.println("Человек  " + name + ": "+ OnDistance );
            }else {
                System.out.println("Человек " + name + ": " + isOnDistance );
            }
        }


        @Override
        public void info() {
            System.out.println("Человек " + name + " MAX пробежал " + Limit + " км" + " и MAX прыгнул " + Power + " м");
        }

}
