package test;

public class Human implements Participant {

    private String name;
    private int limit;
    private int power;

    public Human(String name, int limit, int power) {
        this.name = name;
        this.limit = limit;
        this.power = power;
    }


    @Override
    public int run(int length) {
        return limit;
    }

    @Override
    public int jump(int height) {
        return power;
    }
    @Override
    public void info() {
        System.out.println("Человек " + name + " MAX пробежал " + limit + " км" + " и MAX прыгнул " + power + " м");
    }

    @Override
    public void conclusion() {
        System.out.println("Человек прошел кросс");
    }
}
