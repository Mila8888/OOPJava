package task1;

public class CatClass implements Participants {

    private String name;
    private int Limit;
    private int Power;

    public CatClass(String name, int limit, int power) {
        this.name = name;
        Limit = limit;
        Power = power;
    }

    @Override
    public String name() {
        return null;
    }

    @Override
    public void run(int length) {
        if (length <= Limit) {
            System.out.println("Кот " + name + ": "+ OnDistance );
        }else {
            System.out.println("Кот " + name + ": " + isOnDistance );
        }
    }

    @Override
    public void jump(int height) {
        if (height <= Power) {
            System.out.println("Кот  " + name + ": "+ OnDistance );
        }else {
            System.out.println("Кот " + name + ": " + isOnDistance );
        }
    }


    @Override
    public void info() {
        System.out.println("Кот " + name + " MAX пробежал " + Limit + " км" + " и MAX прыгнул " + Power + " м");
    }
}
